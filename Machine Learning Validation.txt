# ==========================================================
# MultiEvo Machine-Learning Validation Pipeline
# ==========================================================

import pandas as pd
import numpy as np

from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import (
    accuracy_score,
    f1_score,
    confusion_matrix,
    classification_report,
    roc_auc_score
)

from sklearn.preprocessing import label_binarize

import matplotlib.pyplot as plt
import joblib
import umap


# ==========================================================
# 1. Load MultiEvo regulatory dataset
# ==========================================================

data = pd.read_csv(
    "MultiEvo_ML_with_clusters.csv"
)

print(data.shape)
print(data.head())


# ==========================================================
# 2. Define MultiEvo regulatory states
# ==========================================================

data["EPS_state"] = (
    data["EPS"] >
    data["EPS"].median()
)

data["RDS_state"] = (
    data["RDS"] >
    data["RDS"].median()
)


def assign_state(row):

    if row["EPS_state"] and row["RDS_state"]:
        return "High_EPS_High_RDS"

    elif row["EPS_state"] and not row["RDS_state"]:
        return "High_EPS_Low_RDS"

    elif not row["EPS_state"] and row["RDS_state"]:
        return "Low_EPS_High_RDS"

    else:
        return "Low_EPS_Low_RDS"


data["MultiEvo_State"] = data.apply(
    assign_state,
    axis=1
)


print(
    data["MultiEvo_State"].value_counts()
)


# ==========================================================
# 3. Prepare ML features
# ==========================================================

X = data[
    [
        "EPS",
        "RDS"
    ]
]


y = data[
    "MultiEvo_State"
]


# ==========================================================
# 4. Train-test split
# ==========================================================

X_train, X_test, y_train, y_test = train_test_split(
    X,
    y,
    test_size=0.2,
    random_state=123,
    stratify=y
)


# ==========================================================
# 5. Random Forest model
# ==========================================================

rf_state = RandomForestClassifier(
    n_estimators=500,
    random_state=123,
    class_weight="balanced",
    n_jobs=-1
)


rf_state.fit(
    X_train,
    y_train
)


# ==========================================================
# 6. Prediction performance
# ==========================================================

y_pred = rf_state.predict(
    X_test
)


accuracy = accuracy_score(
    y_test,
    y_pred
)


macro_f1 = f1_score(
    y_test,
    y_pred,
    average="macro"
)


print(
    "Accuracy:",
    accuracy
)

print(
    "Macro F1:",
    macro_f1
)


print(
    classification_report(
        y_test,
        y_pred
    )
)


# ==========================================================
# 7. Confusion matrix
# ==========================================================

cm = confusion_matrix(
    y_test,
    y_pred
)


print(cm)


np.savetxt(
    "MultiEvo_confusion_matrix.csv",
    cm,
    delimiter=",",
    fmt="%d"
)



# ==========================================================
# 8. AUROC
# ==========================================================

classes = rf_state.classes_

y_test_bin = label_binarize(
    y_test,
    classes=classes
)


y_prob = rf_state.predict_proba(
    X_test
)


auc = roc_auc_score(
    y_test_bin,
    y_prob,
    multi_class="ovr",
    average="macro"
)


print(
    "Macro AUROC:",
    auc
)



# ==========================================================
# 9. Feature importance
# ==========================================================

feature_importance = pd.DataFrame({

    "Feature":[
        "EPS",
        "RDS"
    ],

    "Importance":
    rf_state.feature_importances_

})


print(feature_importance)


feature_importance.to_csv(
    "MultiEvo_feature_importance.csv",
    index=False
)



# ==========================================================
# 10. Save model
# ==========================================================

joblib.dump(
    rf_state,
    "MultiEvo_RandomForest_model.pkl"
)



# ==========================================================
# 11. Load PCA30 data for UMAP
# ==========================================================

pca30 = pd.read_csv(
    "MultiEvo_PCA30_ML_coordinates.csv"
)


pcs = [
    f"X{i}"
    for i in range(1,31)
]


X30 = pca30[pcs].values



# ==========================================================
# 12. Run UMAP
# ==========================================================

reducer = umap.UMAP(
    n_neighbors=30,
    min_dist=0.3,
    random_state=123
)


umap_coordinates = reducer.fit_transform(
    X30
)


pca30["UMAP1"] = umap_coordinates[:,0]
pca30["UMAP2"] = umap_coordinates[:,1]



# ==========================================================
# 13. Add MultiEvo states
# ==========================================================

umap_state = pca30.merge(
    data[
        [
            "Cell",
            "MultiEvo_State"
        ]
    ],
    on="Cell",
    how="inner"
)


print(
    umap_state.shape
)



# ==========================================================
# 14. UMAP plot
# ==========================================================

plt.figure(
    figsize=(8,7)
)


state_codes = (
    umap_state["MultiEvo_State"]
    .astype("category")
    .cat.codes
)


plt.scatter(
    umap_state["UMAP1"],
    umap_state["UMAP2"],
    c=state_codes,
    s=10,
    alpha=0.6
)


plt.xlabel("UMAP1")
plt.ylabel("UMAP2")

plt.title(
    "MultiEvo Regulatory State Landscape"
)


plt.tight_layout()


plt.savefig(
    "MultiEvo_UMAP_30PC_final.png",
    dpi=300,
    bbox_inches="tight"
)


plt.show()


print("MultiEvo ML validation completed")
